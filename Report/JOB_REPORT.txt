
CREATE TABLE dbo.CL_REPORT_JOB  ( 
	REPORT_JOB_ID       	UNSIGNED BIGINT IDENTITY NOT NULL,
	REPORT_JOB_NAME     	varchar(100) NOT NULL,
	REPORT_JOB_TYPE     	smallint NOT NULL,
	REPORT_JOB_START_DTM	datetime NOT NULL,
	REPORT_JOB_END_DTM  	datetime NULL,
	REPORT_JOB_FILE_NAME	varchar(100) NOT NULL,
	REPORT_JOB_STATUS   	smallint NOT NULL,
	REPORT_JOB_CONDITION	varchar(1000) NOT NULL,
	CREATED             	datetime NOT NULL,
	CREATED_BY          	varchar(50) NOT NULL,
	LAST_UPD            	bigtime NULL,
	LAST_UPD_BY         	varchar(50) NULL,
	NOTIFY_EMAILS       	varchar(1024) NULL 
	)
LOCK ALLPAGES
GO
